function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); // プレイヤー描画 ctx.fillStyle = "lime"; ctx.fillRect(player.x, player.y, player.size, player.size); // 敵処理 enemies.forEach(enemy => { if (player.x > enemy.x) enemy.x += enemy.speed; if (player.x < enemy.x) enemy.x -= enemy.speed; if (player.y > enemy.y) enemy.y += enemy.speed; if (player.y < enemy.y) enemy.y -= enemy.speed; ctx.fillStyle = "red"; ctx.fillRect(enemy.x, enemy.y, enemy.size, enemy.size); // 当たり判定 if ( player.x < enemy.x + enemy.size && player.x + player.size > enemy.x && player.y < enemy.y + enemy.size && player.y + player.size > enemy.y ) { gameOver = true; } }); // スコア描画 ctx.fillStyle = "white"; ctx.font = "50px Arial"; ctx.fillText("Score: " + score, 20, 60); // スコアが30に達したらクリア if (score >= 30) { gameClear = true; } // ゲームオーバー / クリア画面の文字を上書き if (gameOver) { ctx.fillStyle = "white"; ctx.font = "80px serif"; ctx.fillText("どんまい 笑", canvas.width/2 - 200, canvas.height/2); } else if (gameClear) { ctx.fillStyle = "yellow"; ctx.font = "80px serif"; ctx.fillText("クリア！おめでとう！", canvas.width/2 - 400, canvas.height/2); } requestAnimationFrame(draw); }
